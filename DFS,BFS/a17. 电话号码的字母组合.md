class Solution {
    public List<String> letterCombinations(String digits) {
        if(digits.length() == 0) return new ArrayList<String>();
        String[] dic = new String[]{"abc","def","ghi","jkl","mno","pqrs","tuv","wxyz"};
        List<String> ans = new ArrayList<>();
        ans.add("");//注意！
        for(char digit:digits.toCharArray()){//2
            char[] digitChars = dic[Integer.parseInt(Character.toString(digit)) - 2].toCharArray();//abc
            List<String> newAns = new ArrayList<>();
            for(char digitChar : digitChars){//a
                for(String comb:ans){//''
                    String tmp = comb + Character.toString(digitChar);//a
                    newAns.add(tmp);
                }
            }
            ans = newAns;
        }
        return ans;
    }
}
