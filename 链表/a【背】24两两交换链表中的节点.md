class Solution {
    public ListNode swapPairs(ListNode head) {
        ListNode p = new ListNode();
        p.next = head;

        ListNode dummy = p;
        while(p.next != null && p.next.next != null){
            ListNode a = p.next;
            ListNode b = a.next;
            p.next = b;
            a.next = b.next;
            b.next = a;
            p = a;
        }
        return dummy.next;
    }
}
